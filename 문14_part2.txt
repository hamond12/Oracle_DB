SELECT *
FROM DEPT D; --부서

SELECT *
FROM EMP E; --사원

--Q1
SELECT E.ENAME AS 사원이름, D.DNAME AS 사원업무
FROM EMP E, DEPT D 
WHERE E.DEPTNO = D.DEPTNO;

--Q2
SELECT E.ENAME AS 사원이름, E.SAL 
FROM EMP E, DEPT D 
WHERE E.DEPTNO = D.DEPTNO
AND D.DEPTNO = 30;

--Q3
SELECT ENAME, SAL, SAL*0.1 AS 증가액, SAL+SAL*0.1 AS 인상된_급여
FROM EMP 
ORDER BY EMPNO;

--Q4
SELECT  ENAME , DEPTNO 
FROM EMP 
WHERE ENAME LIKE 'S%'

--Q5
SELECT MAX(SAL), MIN(SAL), SUM(SAL), ROUND(AVG(SAL))  
FROM EMP;

--Q6
SELECT D.DNAME AS 업무, COUNT(*) AS 업무별_사원수 
FROM EMP E, DEPT D 
WHERE E.DEPTNO = D.DEPTNO
GROUP BY D.DNAME;

--Q7
SELECT (MAX(SAL) - MIN(SAL)) AS 차액 
FROM EMP;

--Q8
SELECT COUNT(*) AS 구성원_수, SUM(SAL) AS 합계, ROUND(AVG(SAL)) AS 평균 
FROM EMP
WHERE DEPTNO = 30
ORDER BY DEPTNO;

--Q9
SELECT DEPTNO , ROUND(AVG(SAL)) AS AVGSAL
FROM EMP
GROUP BY DEPTNO
ORDER BY AVGSAL 
LIMIT 1;

--Q10
SELECT E.JOB AS 업무명, ROUND(AVG(SAL)) AS 평균급여
FROM EMP E 
WHERE E.JOB != 'SALESMAN'
GROUP BY E.JOB 
HAVING SUM(E.SAL) >= 3000 
ORDER BY 평균급여 DESC;

--Q11
SELECT COUNT(*) AS 직속상관이_있는_사원_수
FROM EMP 
WHERE NOT MGR IS NULL;

-Q12
SELECT ENAME AS 이름, SAL AS 금액, COMM AS 커미션, (SAL + COMM) AS 총액
FROM EMP E 
WHERE COMM IS NOT NULL
ORDER BY 총액 DESC;

--Q13
SELECT DEPTNO AS 부서번호, JOB AS 업무_이름, COUNT(*) AS 인원_수
FROM EMP 
GROUP BY JOB;

--Q14
SELECT D.DNAME 
FROM EMP E RIGHT JOIN DEPT D 
ON E.DEPTNO = D.DEPTNO 
WHERE E.EMPNO IS NULL;

--Q15
SELECT JOB AS 업무 , COUNT(*) AS 인원_수
FROM EMP
GROUP BY JOB 
HAVING COUNT(*) >= 4;

--Q16
SELECT ENAME 
FROM EMP
WHERE EMPNO BETWEEN 7400 AND 7600;

--Q17
SELECT E.ENAME AS 이름, D.DNAME 
FROM EMP E, DEPT D 
WHERE E.DEPTNO = D.DEPTNO ;

--Q18
SELECT ENAME AS 사원, MGR AS 팀장
FROM EMP

--Q19
SELECT E.ENAME AS 이름
FROM EMP E
WHERE E.SAL > (SELECT SAL 
FROM EMP E 
WHERE ENAME = 'SCOTT');
